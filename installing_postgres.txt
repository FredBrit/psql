### Установка одноузлового постгреса без докера
# Обновляем список пакетов
sudo apt update

# Устанавливаем PostgreSQL и дополнительные утилиты
sudo apt install -y postgresql postgresql-contrib

# Проверяем статус постгреса
sudo systemctl status postgresql

# Создаем, пользователя, базу данных и накидываем права
sudo -u postgres psql

CREATE USER fastapi_user WITH PASSWORD 'ваш_сложный_пароль';

CREATE DATABASE myapp_db OWNER fastapi_user;

GRANT ALL ON SCHEMA public TO fastapi_user;

# Проверяем подключение
psql -h 127.0.0.1 -U fastapi_user -d myapp_db

# Чтобы открыть для подключения к базе не только локальные соединения, нужно добавить правило в pg-hba.conf
sudo nano /etc/postgresql/16/main/pg_hba.conf
# IPv6 local connections:
host    all             all             ::1/128                 scram-sha-256
# Внешние подключения
host    all             all             0.0.0.0/0               scram-sha-256

# Проверяем порты и алреса, которые слушает постгрес
sudo netstat -tlnp | grep 5432

# Рестарт и можно пользоваться
sudo systemctl restart postgresql


### Полезные команды

# Зайти в psql под postgres
sudo -u postgres psql

# Посмотреть список баз
\l (внутри psql)

# Посмотреть список пользователей
\du (внутри psql)

# Посмотреть размер базы
SELECT pg_size_pretty(pg_database_size('myapp_db'));

# Логи PostgreSQL
sudo tail -f /var/log/postgresql/postgresql-16-main.log

# Рестарт
sudo systemctl restart postgresql

# Статус
sudo systemctl status postgresql


# Войти в postgres
